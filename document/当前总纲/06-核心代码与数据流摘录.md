function refreshDisplayScreens() {
  if (!sceneReady.value || !splineApp) return

  for (const name of SCREEN_OBJECT_NAMES) {
    const binding = ensureScreenBinding(name)
    if (!binding) continue

    if (!binding.rawObject) {
      binding.rawObject = rawObjectByName(name)
    }

    const changed = drawDisplaySurface(binding, displayPayload.value[name])
    if (binding.rawObject && (changed || !binding.applied)) {
      binding.applied = applyDisplayMaterial(
        binding.rawObject,
        name,
        binding.canvas.toDataURL('image/png'),
        SCREEN_STYLES[name].accent
      )
    }
  }
}

来源: `vite.config.js` + `src/components/SplineScene.vue`

```js
export default defineConfig({
  assetsInclude: ['**/*.splinecode'],
})

import { Application } from '../../vendor/spline-runtime/build/runtime.js'
import sceneUrl from '../../resource/scene.splinecode?url'

// template: <canvas ref="canvasRef" class="spline-canvas" />
onMounted(async () => {
  if (!canvasRef.value) return

  splineApp = new Application(canvasRef.value, {
    renderMode: 'continuous',
  })

  await splineApp.load(sceneUrl)
  sceneReady.value = true

  await applyQueryConfig()
  refreshDisplayScreens()
  refreshIndicatorPanel()
  getDiagnostics()
})
```

// getDiagnostics() 内部核心统计
diagnostics.objectCount = objects.length

diagnostics.hitSource = hoverState.hitSource

async function collectSceneDiagnostics(page) {
  return page.evaluate(async () => {
    const bridge = window.__VISUAL_INSPECTOR__
    if (!bridge) return { bridgeAvailable: false }
    if (bridge.whenReady) await bridge.whenReady()
    return bridge.getDiagnostics ? bridge.getDiagnostics() : { bridgeAvailable: true }
  })
}
